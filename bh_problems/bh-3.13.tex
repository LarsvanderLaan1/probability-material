\begin{exercise}[BH.13]
\begin{solution}
	We have
	\begin{align*}
		P(X=a)& = \sum_{z}P(X=a|Z=z)P(Z=z)\tag*{(\text{LOTP})}\\
		& =  \sum_{z}P(Y=a|Z=z)P(Z=z) \tag*{(\text{Given in the question})}\\
		& =  P(Y=a).\tag*{(\text{Reversed LOTP})}
	\end{align*}
\end{solution}
\end{exercise}
